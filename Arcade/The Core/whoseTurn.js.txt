r = {'a' :8 , 'b':7, 'c':6 , 'd' : 5 , 'e' : 4 , 'f':3 , 'g' : 2 , 'h' : 1}

function whoseTurn(p) {
    
    x = p.split`;`
    
    po1 = (r[x[0][0]] + +x[0][1])%2 ==0 ? 2 : 1
    po2 = (r[x[1][0]] + +x[1][1])%2 ==0 ? 1 : 2
    po3 = (r[x[2][0]] + +x[2][1])%2 ==0 ? 1 : 2
    po4 = (r[x[3][0]] + +x[3][1])%2 ==0 ? 2 : 1
   
    return Math.abs(po1+po2-po3-po4) %2 <1
}
